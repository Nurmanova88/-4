#include <iostream>
using namespace std;

int main() {
    setlocale(LC_ALL, "RU");
    int userNumber;
    int programMultiplier = 1;
    int userAnswer;
    int correctAnswer;

    cout << "[ + ] Введите число: ";
    cin >> userNumber;

    while (programMultiplier <= 9) {
        setlocale(LC_ALL, "RU");
       
        cout << "[ + ] Решите пример: " << userNumber << " x " << programMultiplier << endl;

        cout << "[ + ] Ответ: ";
        cin >> userAnswer;

       
        correctAnswer = userNumber * programMultiplier;

        if (userAnswer == correctAnswer) {
            setlocale(LC_ALL, "RU");
            cout << "[ + ] Пример решен правильно!" << endl;
            programMultiplier++;
        }
        else {
            cout << "[ - ] Ошибка, пример решен неверно!" << endl;
            return 0; 
        }
    }

    cout << "[ + ] Примеры решены, поздравляем!" << endl;

    return 0;
}
